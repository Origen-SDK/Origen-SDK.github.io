def expand_path(path)
  path.split('.').each do |p|
    if p =~ /(.*)\[(\d+)(:(\d+))?\]$/
      msb = Regexp.last_match(2).to_i
      lsb = Regexp.last_match(4).to_i if Regexp.last_match(4)
      yield Regexp.last_match(1)
      if lsb
        yield '[]', msb..lsb
      else
        yield '[]', msb
      end
    else
      yield p
    end
  end
end
