def on_run_flag(node)
  children = node.children.dup
  name = children.shift
  state = children.shift
  unless name.is_a?(Array)
    if @run_flag_table[name.to_sym].nil?
      @run_flag_table[name.to_sym] = 1
    else
      @run_flag_table[name.to_sym] += 1
    end
  end
end
