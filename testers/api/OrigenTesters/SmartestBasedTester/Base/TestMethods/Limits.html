def to_s
  if !lo_limit && !hi_limit
    "\"#{test_name}\"" + ' = "":"NA":"":"NA":"":"":""'
  elsif !lo_limit
    "\"#{test_name}\"" + " = \"\":\"NA\":\"#{hi_limit}\":\"LE\":\"\":\"\":\"0\""
  elsif !hi_limit
    "\"#{test_name}\"" + " = \"#{lo_limit}\":\"GE\":\"\":\"NA\":\"\":\"\":\"0\""
  else
    "\"#{test_name}\"" + " = \"#{lo_limit}\":\"GE\":\"#{hi_limit}\":\"LE\":\"\":\"\":\"0\""
  end
end
