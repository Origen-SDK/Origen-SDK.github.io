def write_spec_yaml(options = {})
  path = options[:approved] ? spec_yaml_approved : spec_yaml_output
  unless Dir.exist?(File.dirname(path))
    FileUtils.mkdir_p(File.dirname(path))
  end
  File.open(path, 'w').puts(to_spec_yaml(options))
  path
end
