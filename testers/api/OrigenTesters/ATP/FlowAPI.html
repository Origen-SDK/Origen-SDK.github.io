def limit(value, options)
  unless options[:rule]
    fail 'You must supply option :rule (e.g. rule: :gt) when calling the limit helper'
  end
  {
    value:    value,
    rule:     options[:rule] || :lt,
    units:    options[:units],
    selector: options[:selector] || options[:test_mode]
  }
end
