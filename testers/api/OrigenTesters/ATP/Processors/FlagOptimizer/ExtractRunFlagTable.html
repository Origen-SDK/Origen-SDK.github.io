def on_if_flag(node)
  children = node.children.dup
  names = children.shift
  state = node.type == :if_flag
  Array(names).each do |name|
    if @run_flag_table[name.to_sym].nil?
      @run_flag_table[name.to_sym] = 1
    else
      @run_flag_table[name.to_sym] += 1
    end
  end
  process_all(node.children)
end
