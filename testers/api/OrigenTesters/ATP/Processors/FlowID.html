def on_id(node)

  end

def on_if_failed(node)
  tid, *nodes = *node
  if tid.is_a?(Array)
    tid = tid.map do |tid|
      if tid =~ /^extern/
        tid
      else
        "#{tid}_#{id}"
      end
    end
  else
    if tid !~ /^extern/
      tid = "#{tid}_#{id}"
    end
  end
  node.updated(nil, [tid] + process_all(nodes))
end
