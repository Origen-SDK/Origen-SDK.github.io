def self.suitable_decompiler_for(pattern: nil, tester: nil, **options)
  if pattern && (Pathname(pattern).extname == '.atp')
    OrigenTesters::IGXLBasedTester
  elsif tester && (tester == 'j750' || tester == 'uflex' || tester == 'ultraflex')
    OrigenTesters::IGXLBasedTester
  end
end
