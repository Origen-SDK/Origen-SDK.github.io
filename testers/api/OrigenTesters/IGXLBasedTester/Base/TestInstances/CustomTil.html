def method_missing(method, *args, &block)
  if definitions[method]
    name = args.shift
    ti = platform::CustomTestInstance.new name, methods: definitions[method].dup,
                                                attrs:   (args.first || {}),
                                                type:    method,
                                                library: self
    test_instances.add(nil, ti)
    ti
  else
    super
  end
end
