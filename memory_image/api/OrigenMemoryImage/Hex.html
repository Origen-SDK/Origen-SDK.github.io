def self.match?(snippet)
  snippet.any? do |line|
    # Match a line like:
    # @180000F0
    line =~ /^@[0-9a-fA-F]+\s?$/
  end
end

def start_address

    lines.each do |line|
      if line =~ /^@([0-9a-fA-F]+)\s?$/
        return Regexp.last_match[1].to_i(16)

    end
  end
