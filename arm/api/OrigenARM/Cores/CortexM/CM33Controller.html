def enter_debug_mode(halt_core: true, **options)
  pp('Entering Debug Mode...') do
    reg(:dhcsr).bits(:dbgkey).write(OrigenARM::Cores::CortexM::CM33::Registers::DHCSR_DBGKEY)
    reg(:dhcsr).bits(:c_debugen).write(1)
    reg(:dhcsr).write!
    
    enter_debug_mode_delay!

    if halt_core
      reg(:dhcsr).bits(:dbgkey).write(OrigenARM::Cores::CortexM::CM33::Registers::DHCSR_DBGKEY)
      reg(:dhcsr).bits(:c_debugen).write(1)
      reg(:dhcsr).bits(:c_halt).write(1)
      reg(:dhcsr).write!
      enter_debug_mode_delay!
    end
  end
end

def exit_debug_mode(release_core: true, **options)
  pp('Exiting Debug Mode...') do
    reg(:dhcsr).bits(:dbgkey).write(OrigenARM::Cores::CortexM::CM33::Registers::DHCSR_DBGKEY)
    reg(:dhcsr).bits(:c_halt).write(0) if release_core
    reg(:dhcsr).bits(:c_debugen).write(0)
    reg(:dhcsr).write!
    
    cc 'Delay for the core to exit debug mode'
    exit_debug_mode_delay!
  end
end

def initialize_core(pc:, sp:, release_core: false, disable_security: false, sp_lower_limit: nil, sp_upper_limit: nil)
  # As cores are added to K4, build up support for them here.
  enter_debug_mode
  set_sp(sp)
  set_pc(pc)
  disable_security if disable_security
  set_stack_limits(sp_lower_limit, sp_upper_limit) if (sp_lower_limit || sp_upper_limit)
  exit_debug_mode(release_core: release_core)
end

Sets the program counter. This involves two steps:

1. Writing the THUMB bit to enable core execution.
2. Writing the debug return address core register.

On the next instruction,the core will jump to this address upon execution start/restart

def set_pc(pc, options={})
  pp('Patch the Program Counter') do
    # Force the thumb bit. Nothing will work otherwise as CM33 only support THUMB
    reg(:xpsr).bits(:t).write(1)
    #write_debug_register!(:xpsr, xpsr, options)
    reg(:xpsr).write!
    
    # Write the debug return address with the new PC
    # Add 1 to indicate thumb mode
    #write_debug_register!(:dbg_retn_addr, pc + 1, options)
    #reg(:dbg_retn_addr).write!(pc + 1)
    reg(:debug_return_address).write!(pc + 1)
  end
end
