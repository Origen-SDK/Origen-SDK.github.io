def _bit_in_range?(bits, max, min)
  upper = bits.position + bits.size - 1
  lower = bits.position
  !((lower > max) || (upper < min))
end

def _bit_rw(bits)
  str = ''
  if bits.readable?
    str += 'readable'
  else
    str += 'not-readable'
  end
  if bits.writable?
    str += ' writable'
  else
    str += ' not-writable'
  end
  str.strip
end

def _num_bits_in_range(bits, max, min)
  upper = bits.position + bits.size - 1
  lower = bits.position
  [upper, max].min - [lower, min].max + 1
end
