def [](*args)
  if args.length == 1 && !args.first.is_a?(Range)
    old_bit_select(args.first)
  else
    if args.first.is_a?(Range)
      msb = args.first.first
      lsb = args.first.last
    else
      msb = args.first
      lsb = args.last
    end
    (self >> lsb) & 0.ones_comp(msb - lsb + 1)
  end
end

def cycles
  if block_given?
    times do
      yield
      Origen.app.tester.cycle
    end
  else
    Origen.app.tester.cycle(repeat: self)
  end
end
