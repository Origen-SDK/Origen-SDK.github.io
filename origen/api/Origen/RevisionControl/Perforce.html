def initialize(options = {})
  super
  @remotes_method = :print
  @p4 = P4.new
  @p4.maxresults = 1
  parse_remote
  @p4.connect
  unless @p4.connected?
    Origen.log.error("Could not connect to port #{@p4.port} on client #{@p4.client}!")
    fail
  end
  @p4.run_login
end

def print(options = {})
  options = {
    verbose: true,
    version: 'Latest'
  }.merge(options)
  cmd = [__method__.to_s, '-o', "#{@local}/#{@remote_file.to_s.split('/')[-1]}", @remote_file.to_s]
  run cmd
end
