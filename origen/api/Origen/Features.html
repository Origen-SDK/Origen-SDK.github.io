def feature(name = nil)
  if !name
    self.class.features.keys
  else
    if self.class.features.key?(name)
      self.class.features[name]
    else
      fail "Feature #{name} does not exist!"
    end
  end
end

def has_features?(name = nil)
  if !name
    if feature.size == 0
      return false
    else
      return true
    end
  else
    return feature.include?(name)
  end
end
