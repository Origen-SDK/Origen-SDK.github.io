def bug(name, options = {})
  name = name.to_s.downcase.to_sym
  bugs[name] = Bug.new(name, options)
  define_method "has_#{name}_bug?" do
    Origen.deprecate "Use has_bug?(:#{name}) instead of has_#{name}_bug?"
    has_bug?(name)
  end
end
