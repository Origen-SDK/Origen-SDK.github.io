def fixed_on_version

      affected_versions.max + 1
    end
  end

def present_on_version?(version, _options = {})
  if affected_versions.empty?
    if fixed_on_version
      version < fixed_on_version
    else
      true
    end
  else
    affected_versions.include?(version)
  end
end
