def find_mode(name)
  name = name.to_s.downcase.to_sym
  if MODES.include?(name)
    name
  else
    mode = MODES.find do |m|
      m.to_s =~ /^#{name}/
    end
    if mode
      mode
    else
      fail "Invalid mode requested, must be one of: #{MODES}"
    end
  end
end
