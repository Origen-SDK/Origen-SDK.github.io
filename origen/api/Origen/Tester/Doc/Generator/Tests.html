def add(name, attrs = {})
  test = Test.new(name, attrs)
  if @current_group
    @current_group << test
  else
    collection << test
  end
  test
end

def group(name = nil, options = {})
  name, options = nil, name if name.is_a?(Hash)
  @current_group = TestGroup.new(name, options)
  collection << @current_group
  yield @current_group
  @current_group = nil
end
