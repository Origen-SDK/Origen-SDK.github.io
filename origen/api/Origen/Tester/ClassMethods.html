def new(*args, &block) # :nodoc:
  if Origen.app.with_doc_tester?
    x = Origen::Tester::Doc.allocate
    if Origen.app.with_html_doc_tester?
      x.html_mode = true
    end
  else
    x = allocate
  end
  x.send(:initialize, *args, &block)
  x.register_tester
  x
end
