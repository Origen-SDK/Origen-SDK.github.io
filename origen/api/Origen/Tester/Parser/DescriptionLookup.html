def add_for_flow(flow, description, _options = {})
  k = flow_key(flow)
  @store[:flow][k] ||= []
  [description].flatten.each do |d|
    @store[:flow][k] << d
  end
end

def add_for_test_usage(test, flow, description, _option = {})
  k = flow_key(flow)
  n = name_key(test)
  @store[:usage][k] ||= {}
  @store[:usage][k][n] ||= []
  [description].flatten.each do |d|
    @store[:usage][k][n] << d
  end
end
