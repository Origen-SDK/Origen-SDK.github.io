def parameters_context(obj = nil)
  if obj
    if obj.is_a?(Symbol)
      valid = [:top, :dut].include?(obj)
    end
    valid ||= obj.is_a?(String)
    unless valid
      fail 'Invalid parameters context, must be :top or a string path to a model object'
    end
    @parameters_context = obj
  else
    @parameters_context
  end
end
