def origen_subblock?
  self.is_a?(Origen::Model) || self.is_a?(Origen::Controller) || self.is_a?(Origen::SubBlocks::Placeholder)
end

def try(*methods)
  methods.each do |method|
    if self.respond_to?(method)
      val = send(method)
      return val if val
    end
  end
  nil
end
