def cycles
  if block_given?
    times do
      yield
      Origen.app.tester.cycle
    end
  else
    Origen.app.tester.cycle(repeat: self)
  end
end

def to_bool
  if self == 1
    return true
  elsif self == 0
    return false
  else
    return nil
  end
end

def to_spreadsheet_column

  index_hash[self]
end
